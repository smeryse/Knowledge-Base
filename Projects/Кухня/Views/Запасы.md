## Активные запасы

```dataviewjs
const pantry = dv.pages('"Projects/Кухня/Pantry"')
  .where(p => p.type === "pantry-item" && p.status != "consumed" && p.status != "discarded");

function deriveExpiry(item) {
  const manufacturedOn = item.manufactured_on ? dv.date(item.manufactured_on) : null;
  const product = item.product?.path ? dv.page(item.product.path) : null;
  const shelfLifeDays = Number(product?.default_shelf_life_days || 0);
  if (!manufacturedOn || !shelfLifeDays) return null;
  return manufacturedOn.plus({ days: shelfLifeDays });
}

const rows = pantry
  .map(item => ({
    product: item.product,
    qty: item.qty_current ?? "-",
    unit: item.unit ?? "-",
    manufacturedOn: item.manufactured_on || "-",
    expiresOn: deriveExpiry(item)
  }))
  .sort(row => row.expiresOn ? row.expiresOn.toMillis() : Number.MAX_SAFE_INTEGER)
  .map(row => [row.product, row.qty, row.unit, row.manufacturedOn, row.expiresOn ?? "-"]);

dv.table(["Товар", "Остаток", "Ед.", "Изготовлено", "Годен до"], rows);
```

## Скоро испортится

```dataviewjs
const pantry = dv.pages('"Projects/Кухня/Pantry"')
  .where(p => p.type === "pantry-item" && p.status != "consumed" && p.status != "discarded");

function deriveExpiry(item) {
  const manufacturedOn = item.manufactured_on ? dv.date(item.manufactured_on) : null;
  const product = item.product?.path ? dv.page(item.product.path) : null;
  const shelfLifeDays = Number(product?.default_shelf_life_days || 0);
  if (!manufacturedOn || !shelfLifeDays) return null;
  return manufacturedOn.plus({ days: shelfLifeDays });
}

const today = dv.date('today');
const rows = [];

for (const item of pantry) {
  const expiresOn = deriveExpiry(item);
  const daysLeft = expiresOn ? Math.round(expiresOn.diff(today, 'days').days) : null;
  if (daysLeft !== null && daysLeft <= 5) {
    rows.push([
      item.product,
      item.qty_current ?? "-",
      item.unit ?? "-",
      item.manufactured_on ?? "-",
      expiresOn,
      daysLeft
    ]);
  }
}

rows.sort((a, b) => a[4].toMillis() - b[4].toMillis());
dv.table(["Товар", "Остаток", "Ед.", "Изготовлено", "Годен до", "Дней осталось"], rows);
```
