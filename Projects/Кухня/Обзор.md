# Обзор

## Сводка

```dataviewjs
const products = dv.pages('"Projects/Кухня/Products"').where(p => p.type === "product");
const recipes = dv.pages('"Projects/Кухня/Recipes"').where(p => p.type === "recipe" && p.recipe_status !== "archived");
const receipts = dv.pages('"Projects/Кухня/Receipts"').where(p => p.type === "receipt");
const receiptItems = dv.pages('"Projects/Кухня/Receipt Items"').where(p => p.type === "receipt-item");
const pantry = dv.pages('"Projects/Кухня/Pantry"').where(p => p.type === "pantry-item");
const cooking = dv.pages('"Projects/Кухня/Cooking Log"').where(p => p.type === "cooking-entry");
const mealPlans = dv.pages('"Projects/Кухня/Meal Plans"').where(p => p.type === "meal-plan");
const shopping = dv.pages('"Projects/Кухня/Shopping List"').where(p => p.type === "shopping-item" && p.status != "done" && p.status != "cancelled");

function deriveExpiry(item) {
  const manufacturedOn = item.manufactured_on ? dv.date(item.manufactured_on) : null;
  const product = item.product?.path ? dv.page(item.product.path) : null;
  const shelfLifeDays = Number(product?.default_shelf_life_days || 0);
  if (!manufacturedOn || !shelfLifeDays) return null;
  return manufacturedOn.plus({ days: shelfLifeDays });
}

const today = dv.date('today');
const expiringSoon = pantry.where(p => {
  const expiresOn = deriveExpiry(p);
  return expiresOn && expiresOn <= today.plus({ days: 3 });
});

dv.table(["Показатель", "Значение"], [
  ["Товаров в базе", products.length],
  ["Рецептов", recipes.length],
  ["Чеков", receipts.length],
  ["Позиций в чеках", receiptItems.length],
  ["Запасов дома", pantry.length],
  ["Приготовлений", cooking.length],
  ["Планов питания", mealPlans.length],
  ["Пунктов к покупке", shopping.length],
  ["Скоро истекают", expiringSoon.length]
]);
```

## Скоро истекает

```dataviewjs
const pantryExpiring = dv.pages('"Projects/Кухня/Pantry"')
  .where(p => p.type === "pantry-item" && p.status != "consumed" && p.status != "discarded");

function deriveExpiry(item) {
  const manufacturedOn = item.manufactured_on ? dv.date(item.manufactured_on) : null;
  const product = item.product?.path ? dv.page(item.product.path) : null;
  const shelfLifeDays = Number(product?.default_shelf_life_days || 0);
  if (!manufacturedOn || !shelfLifeDays) return null;
  return manufacturedOn.plus({ days: shelfLifeDays });
}

const rows = pantryExpiring
  .map(item => ({
    product: item.product,
    qty: item.qty_current ?? "-",
    manufacturedOn: item.manufactured_on ?? "-",
    expiresOn: deriveExpiry(item)
  }))
  .where(row => row.expiresOn)
  .sort(row => row.expiresOn.toMillis())
  .slice(0, 15)
  .map(row => [row.product, row.qty, row.manufacturedOn, row.expiresOn]);

dv.table(["Товар", "Остаток", "Изготовлено", "Годен до"], rows);
```
