# Что купить

## Активный список

```dataview
TABLE product as "Товар", target_qty as "Нужно", unit as "Ед.", priority as "Приоритет", preferred_store as "Предпочтительный магазин", max_target_price as "Лимит цены", notes as "Комментарий"
FROM "Projects/Еда/Shopping List"
WHERE type = "shopping-item" AND status != "done" AND status != "cancelled"
SORT priority DESC, file.mtime DESC
```

## Лучшие известные цены по активному списку

```dataviewjs
const shopping = dv.pages('"Projects/Еда/Shopping List"')
  .where(p => p.type === "shopping-item" && p.status != "done" && p.status != "cancelled" && p.product);

const items = dv.pages('"Projects/Еда/Receipt Items"').where(p => p.type === "receipt-item");

const rows = [];

for (const entry of shopping) {
  const productPath = entry.product?.path;
  const matches = items.where(i => i.product?.path === productPath);
  if (matches.length === 0) {
    rows.push([entry.product, "-", "-", "-", "Нет истории"]);
    continue;
  }

  const sorted = [...matches].sort((a, b) => {
    const aValue = a.price_per_base_unit ?? a.price_total ?? Number.MAX_SAFE_INTEGER;
    const bValue = b.price_per_base_unit ?? b.price_total ?? Number.MAX_SAFE_INTEGER;
    return aValue - bValue;
  });

  const best = sorted[0];
  rows.push([
    entry.product,
    best.store ?? "-",
    best.price_per_base_unit ?? "-",
    best.price_total ?? "-",
    best.date ?? "-"
  ]);
}

dv.table(["Товар", "Лучший магазин", "Цена за баз. ед.", "Цена упаковки", "Дата"], rows);
```

## Купить, если дома пусто

```dataviewjs
const shopping = dv.pages('"Projects/Еда/Shopping List"')
  .where(p => p.type === "shopping-item" && p.status != "done" && p.status != "cancelled" && p.product);
const pantry = dv.pages('"Projects/Еда/Pantry"')
  .where(p => p.type === "pantry-item" && p.status != "consumed" && p.status != "discarded" && p.product);

const rows = [];

for (const entry of shopping) {
  const productPath = entry.product?.path;
  const activePantry = pantry.where(p => p.product?.path === productPath);
  if (activePantry.length === 0) {
    rows.push([entry.product, entry.priority ?? "-", entry.preferred_store ?? "-", entry.notes ?? "-"]);
  }
}

dv.table(["Товар", "Приоритет", "Магазин", "Почему нужен"], rows);
```
