# Прогресс

Статистика выполнения задач и баллов.

---

## 🏆 Баллы за сегодня

```dataviewjs
const today = dateformat(date(today), "yyyy-MM-dd");
const filePath = "Tasks/Daily/" + today + ".md";
const file = dv.page(filePath);
if (file) {
    const content = await dv.io.load(filePath);
    const taskRegex = /-\s?\[([xX ])\].*?\(\+(\d+)\)/g;
    const rubleRegex = /-\s?\[([xX ])\].*?\(\+(\d+)р\)/g;
    let completed = 0, total = 0;
    for (const match of content.matchAll(taskRegex)) {
        total += parseInt(match[2]);
        if (match[1].toLowerCase() === 'x') completed += parseInt(match[2]);
    }
    for (const match of content.matchAll(rubleRegex)) {
        const rublePoints = Math.round(parseInt(match[2]) * 0.05);
        total += rublePoints;
        if (match[1].toLowerCase() === 'x') completed += rublePoints;
    }
    dv.paragraph(`**Выполнено:** ${completed} / **Всего:** ${total} баллов`);
} else {
    dv.paragraph("*Заметка на сегодня не найдена.*");
}
```

## 📈 За неделю

```dataviewjs
// Сумма баллов по дням текущей недели
const now = new Date();
const dayOfWeek = now.getDay() || 7;
const monday = new Date(now);
monday.setDate(now.getDate() - dayOfWeek + 1);

let weekTotal = 0;
let weekDone = 0;

for (let i = 0; i < 7; i++) {
    const d = new Date(monday);
    d.setDate(monday.getDate() + i);
    const dateStr = d.toISOString().slice(0, 10);
    const path = "Tasks/Daily/" + dateStr + ".md";
    try {
        const content = await dv.io.load(path);
        if (!content) continue;
        const taskRegex = /-\s?\[([xX ])\].*?\(\+(\d+)\)/g;
        const rubleRegex = /-\s?\[([xX ])\].*?\(\+(\d+)р\)/g;
        for (const match of content.matchAll(taskRegex)) {
            weekTotal += parseInt(match[2]);
            if (match[1].toLowerCase() === 'x') weekDone += parseInt(match[2]);
        }
        for (const match of content.matchAll(rubleRegex)) {
            const rp = Math.round(parseInt(match[2]) * 0.05);
            weekTotal += rp;
            if (match[1].toLowerCase() === 'x') weekDone += rp;
        }
    } catch (e) {}
}

dv.paragraph(`**Неделя:** ${weekDone} / ${weekTotal} баллов`);
```

## 📊 По дням (таблица)

```dataviewjs
const days = [];
for (let i = 6; i >= 0; i--) {
    const d = new Date();
    d.setDate(d.getDate() - i);
    days.push(d.toISOString().slice(0, 10));
}

const rows = [];
for (const dateStr of days) {
    const path = "Tasks/Daily/" + dateStr + ".md";
    try {
        const content = await dv.io.load(path);
        if (!content) { rows.push([dateStr, 0, 0]); continue; }
        const taskRegex = /-\s?\[([xX ])\].*?\(\+(\d+)\)/g;
        const rubleRegex = /-\s?\[([xX ])\].*?\(\+(\d+)р\)/g;
        let total = 0, done = 0;
        for (const match of content.matchAll(taskRegex)) {
            total += parseInt(match[2]);
            if (match[1].toLowerCase() === 'x') done += parseInt(match[2]);
        }
        for (const match of content.matchAll(rubleRegex)) {
            const rp = Math.round(parseInt(match[2]) * 0.05);
            total += rp;
            if (match[1].toLowerCase() === 'x') done += rp;
        }
        rows.push([dateStr, done, total]);
    } catch (e) {
        rows.push([dateStr, 0, 0]);
    }
}

dv.table(["Дата", "Выполнено", "Всего"], rows);
```
