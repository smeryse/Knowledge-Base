# 👕 Состояние всей одежды

Обзор всех вещей и их ресурс до стирки.

---

## 📊 Сводка

```dataviewjs
const ITEMS_DIR = 'LifeOS/Одежда/items';
const allFiles = app.vault.getMarkdownFiles().filter(f => f.path.startsWith(ITEMS_DIR));

let total = allFiles.length;
let clean = 0, dirty = 0;
let totalWears = 0, maxWears = 0;

for (const file of allFiles) {
    const content = await app.vault.read(file);
    const yamlMatch = content.match(/^---\n([\s\S]*?)\n---/);
    if (!yamlMatch) continue;
    
    const yaml = {};
    for (const line of yamlMatch[1].split('\n')) {
        const match = line.match(/^(\w+):\s*(.+)$/);
        if (match) {
            let val = match[2].trim();
            if (val.startsWith('"') && val.endsWith('"')) val = val.slice(1, -1);
            if (val === 'true') val = true;
            else if (val === 'false') val = false;
            else if (val === 'null') val = null;
            else if (!isNaN(val) && val !== '') val = Number(val);
            yaml[match[1]] = val;
        }
    }
    
    if (yaml.is_dirty) dirty++;
    else clean++;
    
    totalWears += yaml.wears_count || 0;
    maxWears += yaml.wash_after_wears || 0;
}

const usagePercent = maxWears > 0 ? Math.round((totalWears / maxWears) * 100) : 0;

// Визуальная сводка
dv.paragraph(`**Всего вещей:** ${total}`);
dv.paragraph(`✅ **Чистые:** ${clean} | ⚠️ **Грязные:** ${dirty}`);
dv.paragraph(`**Общий износ:** ${totalWears} носок из ${maxWears} возможных (${usagePercent}%)`);
```

## 📋 Все вещи по категориям

```dataviewjs
const ITEMS_DIR = 'LifeOS/Одежда/items';
const allFiles = app.vault.getMarkdownFiles().filter(f => f.path.startsWith(ITEMS_DIR));

// Группируем по типу
const byType = {};

for (const file of allFiles) {
    const content = await app.vault.read(file);
    const yamlMatch = content.match(/^---\n([\s\S]*?)\n---/);
    if (!yamlMatch) continue;
    
    const yaml = {};
    for (const line of yamlMatch[1].split('\n')) {
        const match = line.match(/^(\w+):\s*(.+)$/);
        if (match) {
            let val = match[2].trim();
            if (val.startsWith('"') && val.endsWith('"')) val = val.slice(1, -1);
            if (val === 'true') val = true;
            else if (val === 'false') val = false;
            else if (val === 'null') val = null;
            else if (!isNaN(val) && val !== '') val = Number(val);
            yaml[match[1]] = val;
        }
    }
    
    const type = yaml.type || 'Другое';
    if (!byType[type]) byType[type] = [];
    byType[type].push(yaml);
}

// Выводим по категориям
for (const [type, items] of Object.entries(byType)) {
    dv.header(3, type);
    
    const rows = items.map(item => {
        const pct = item.wash_after_wears > 0 ? Math.round((item.wears_count / item.wash_after_wears) * 100) : 0;
        const barLen = 10;
        const filled = Math.min(Math.round((pct / 100) * barLen), barLen);
        const bar = '█'.repeat(filled) + '░'.repeat(barLen - filled);
        
        const status = item.is_dirty ? '🔴' : pct >= 80 ? '🟡' : '🟢';
        
        return [
            status,
            item.name,
            item.color,
            `${item.wears_count}/${item.wash_after_wears}`,
            `${bar} ${pct}%`,
            item.weight_str || `${item.weight}г`
        ];
    });
    
    dv.table(["", "Название", "Цвет", "Надето", "Износ", "Вес"], rows);
}
```

## 🔴 Готовые к стирке

```dataview
TABLE color as "Цвет", wears_count as "Надето", wash_after_wears as "Лимит", weight_str as "Вес"
FROM "LifeOS/Одежда/items"
WHERE is_dirty = true
SORT type, name
```

## 🟡 Почти готовы (80%+ ресурса)

```dataview
TABLE color as "Цвет", wears_count as "Надето", wash_after_wears as "Лимит", round(wears_count/wash_after_wears*100) as "Износ %"
FROM "LifeOS/Одежда/items"
WHERE is_dirty = false AND wears_count/wash_after_wears >= 0.8
SORT (wears_count/wash_after_wears) DESC
```
